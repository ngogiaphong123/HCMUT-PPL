
         climateChange : function void (arr : array[4] of integer) {
             n = length(arr);
             for (i = 0, i < n - 1, i + 1) {
                 for (j = 0, j < n - i - 1, j + 1) {
                     if (arr[j] > arr[j + 1]) {
                         temp = arr[j];
                         arr[j] = arr[j + 1];
                         arr[j + 1] = temp;
                     }
                 }
             }
         }
         main: function void () {
             arr = {64, 34, 25, 12, 22, 11, 90};
             climateChange(arr);
             printInt(arr);
         }
