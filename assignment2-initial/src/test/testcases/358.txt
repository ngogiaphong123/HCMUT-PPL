
        main: function integer() {
            arr: array[5] of integer = {3, 7, 9, 2, 4};
            target: integer = 9;
            n: integer = 5;
            found: boolean = false;
            index: integer = -1;
            for (i = 0, i < n, i + 1) {
                if (arr[i] == target) {
                    found = true;
                    index = i;
                    break;
                }
            }
            if (found) {
                return index;
            }
            else {
                return -1;
            }
        }
